\begin{equation}
    \vec{\dot{x}} = A_c\vec{x} + B_c \vec{u} 
\end{equation}

\begin{equation*}
A_c =
    \begin{bmatrix}
        0 &  1 &  0 & 0 \\
        0 &  0 &  -K_2 & 0  \\
        0 &  0 &  0 & 1 \\
        0 &  0 & -K_1K_{pp} & -K_1K_{pd}                                 
    \end{bmatrix}
    , \quad B_c = 
    \begin{bmatrix} 0 \\ 0 \\ 0 \\ K_1K_{pp} \end{bmatrix}
\end{equation*}

\begin{align*}
    \vec{x_{k+1}} &= \vec{x_k} + h\vec{\dot{x}_k} \\
                  &= \vec{x_k} + h(A_c\vec{x_k} + B_c \vec{u_k}) \\  
                  &= (I + hA_C)\vec{x_k} + hB_c \vec{u_k} \\  
\end{align*}
